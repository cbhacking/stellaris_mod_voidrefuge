# Colonize habitats
country_event = {
	id = mod_voidrefuge_game_start.3
	hide_window = yes
	is_triggered_only = yes

	trigger = {
		has_civic = civic_mod_voidrefuge
	}

	immediate = {
		# Set up the capital
		home_planet = {
			# Remove pre-generated pops, buildings, districts
			while = {
				limit = { num_pops > 1 }
				random_pop = {
					kill_pop = yes
				}
			}
			remove_all_buildings = yes
			remove_all_districts = yes
			clear_blockers = yes
			
			# 10 pops need housing (plus some margin) and jobs
			while = {
				limit = { num_pops < 10 }
				create_pop = {
					species = owner_main_species
					ethos = random
				}
			}

			# 6 housing, 0 jobs
			add_district = district_hab_housing

			# 3 housing, 3 jobs
			add_district = district_hab_science

			# 5 housing, 4 jobs
			add_building = building_hab_major_capital

			# 0 housing, 4 jobs
			add_building = building_hydroponics_farm
			add_building = building_foundry_1
		}

		# Set up the mining habitat
		random_planet_within_border = {
			limit = { has_planet_flag = voidrefuge_mining_habitat }
			# create_colony = {
			# 	owner = ROOT
			# 	species = owner_main_species
			# 	ethos = owner
			# }
			# set_name = "[mod_voidrefuge_mining_planet.GetName] Habitat"

			# 6 pops need housing (plus some margin) and jobs
			while = {
				limit = { num_pops < 6 }
				create_pop = {
					species = owner_main_species
					ethos = random
				}
			}
			
			# 6 housing, 6 jobs
			add_district = district_hab_mining
			add_district = district_hab_mining

			# 3 housing, 2 job
			add_building = building_hab_capital

			# 0 housing, 2 jobs
			add_building = building_foundry_1

			check_planet_employment = yes
		}

		# Set up the energy habitat
		random_planet_within_border = {
			limit = { has_planet_flag = voidrefuge_energy_habitat }
			# create_colony = {
			# 	owner = ROOT
			# 	species = owner_main_species
			# 	ethos = owner
			# }
			#set_name = "HABITAT_PLANET_NAME"

			remove_all_buildings = yes
			remove_all_districts = yes

			# 8 pops need housing (plus some margin) and jobs
			while = {
				limit = { num_pops < 8 }
				create_pop = {
					species = owner_main_species
					ethos = random
				}
			}

			# 6 housing, 6 jobs
			add_district = district_hab_energy
			add_district = district_hab_energy

			# 3 housing, 2 job
			add_building = building_hab_capital

			# 0 housing, 2 jobs
			add_building = building_factory_1

			check_planet_employment = yes
		}
	}
}

# Delete and re-create malfunctioning mining stations
country_event = {
	id = mod_voidrefuge_game_start.5
	hide_window = yes
	is_triggered_only = yes

	trigger = { has_civic = civic_mod_voidrefuge }

	immediate = {
		every_mining_station = {
			limit = {
				exists = owner
				owner = { is_country = ROOT }
			}
			delete_fleet = THIS
		}
		country_event = {
			id = game_start.33
		}
	}
}
